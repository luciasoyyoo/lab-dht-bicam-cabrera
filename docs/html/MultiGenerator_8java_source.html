337 public Generator roulette() {
338 float totalWeight = 0;
339 for (int i = 0; i < listGenerators.length; i++) {
340 totalWeight = listGenerators[i].getWeight() + totalWeight;
341 }
342 List<Float> listProb = new ArrayList<Float>();
343 for (int i = 0; i < listGenerators.length; i++) {
344 float probF = listGenerators[i].getWeight() / totalWeight;
345 listProb.add(probF);
346 }
347 List<LimitRoulette> listLimit = new ArrayList<LimitRoulette>();
348 float limitHigh = 0;
349 float limitLow = 0;
350 for (int i = 0; i < listProb.size(); i++) {
351 LimitRoulette limitRoulette = new LimitRoulette();
352 limitHigh = listProb.get(i) + limitHigh;
353 limitRoulette.setLimitHigh(limitHigh);
354 limitRoulette.setLimitLow(limitLow);
355 limitLow = limitHigh;
356 limitRoulette.setGenerator(listGenerators[i]);
357 listLimit.add(limitRoulette);
358 }
359 // Uses ThreadLocalRandom for algorithmic (non-cryptographic) randomness.
360 // This RNG chooses a generator according to roulette weights and is not
361 // used for security-sensitive purposes. Suppress Sonar hotspot S2245.
362 @SuppressWarnings("squid:S2245")
363 float numbAleatory = ThreadLocalRandom.current().nextFloat();
364 boolean find = false;
365 int i = 0;
366 while ((find == false) && (i < listLimit.size())){
367 if((listLimit.get(i).getLimitLow() <= numbAleatory) && (numbAleatory <= listLimit.get(i).getLimitHigh())){
368 find = true;
369 }
370 else i++;
371 }
372 if (find) {
373 return listLimit.get(i).getGenerator();
374 }
375 else return listLimit.get(listLimit.size() - 1).getGenerator();
376 }
