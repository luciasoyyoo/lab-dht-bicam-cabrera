173 public void inicialiceLBest (){
174 for (int j = 0; j < coutSwarm; j++) {
175 // pick initial reference from the particle's personal best
176 State reference = listParticle.get(getCountParticle()).getStatePBest();
177 int iterator = countParticleBySwarm + getCountParticle();
178 if(Strategy.getStrategy().getProblem().getTypeProblem().equals(ProblemType.Maximizar)){
179 for (int i = countParticle; i < iterator; i++) {
180 if (listParticle.get(i).getStatePBest().getEvaluation().get(0) > reference.getEvaluation().get(0))
181 reference = listParticle.get(i).getStatePBest();
182 setCountParticle(getCountParticle() + 1);
183 }

188 reference = listParticle.get(i).getStatePBest();
189 setCountParticle(getCountParticle() + 1);

220 private List<Particle> getListStateRef() {
221 Boolean found = false;
222 List<String> key = Strategy.getStrategy().getListKey();
223 int count = 0;
224 if(RandomSearch.listStateReference.size() == 0){
225 return this.setListParticle(new ArrayList<Particle>());
226 }
227 while((found.equals(false)) && (Strategy.getStrategy().mapGenerators.size() > count)){
228 //recorrer la lista de generadores, hasta que encuentre el PSO
229 if(key.get(count).equals(GeneratorType.ParticleSwarmOptimization.toString())){
230 //creo el generador PSO, y si su lista de particulas esta vacia entonces es la primera vez que lo estoy creando, y cada estado lo convierto en particulas
231 GeneratorType keyGenerator = GeneratorType.valueOf(String.valueOf(key.get(count)));
232 ParticleSwarmOptimization generator = (ParticleSwarmOptimization) Strategy.getStrategy().mapGenerators.get(keyGenerator);
233 if(generator.getListParticle().isEmpty()){
234 //convertir los estados en particulas
235 for (int j = 0; j < RandomSearch.listStateReference.size(); j++) {
236 //si el estado es creado con el generator RandomSearch entonces la convierto en particula
237 if(getListParticle().size() != countRef){
238 ArrayList<Object> velocity = new ArrayList<Object>();
239 State stateAct = (State) RandomSearch.listStateReference.get(j).getCopy();
240 stateAct.setCode(new ArrayList<Object>(RandomSearch.listStateReference.get(j).getCode()));
241 stateAct.setEvaluation(RandomSearch.listStateReference.get(j).getEvaluation());
242
243 State statePBest = (State) RandomSearch.listStateReference.get(j).getCopy();
244 statePBest.setCode(new ArrayList<Object>(RandomSearch.listStateReference.get(j).getCode()));
245 statePBest.setEvaluation(RandomSearch.listStateReference.get(j).getEvaluation());
246
247 Particle particle = new Particle(stateAct, statePBest, velocity);
248 getListParticle().add(particle);
249 }
250 }
251 }
252 else{
253 setListParticle(generator.getListStateReference());
254 }
255 found = true;
256 }
257 count++;
258 }
259 return getListParticle();
260 }
